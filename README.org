* Build

#+begin_src elisp
  (require 'ox-publish)
  (let ((org-publish-project-alist
         '(("blog"
            :base-directory "src"
            :publishing-directory "docs"
            :auto-sitemap t
            :recursive t
            :publishing-function org-html-publish-to-html))))
    (org-publish-all t)
    (message "Site built at %s" (format-time-string "%Y-%m-%d %H:%M:%S")))
#+end_src
